1. Install dependencies:
npm install --save-dev jest ts-jest @types/jest @testing-library/react @testing-library/jest-dom jest-environment-jsdom identity-obj-proxy @types/testing-library__jest-dom jest-fetch-mock

2. Add jest script:
add "test": "jest" to package.json scripts

3. Create jest.config.js:
module.exports = {
  preset: 'ts-jest',
  testEnvironment: 'jsdom',
  setupFilesAfterEnv: ['<rootDir>/jest.setup.js'],
  moduleNameMapper: {
    '^@/(.*)$': '<rootDir>/$1',
    '\\.(css|scss|sass)$': 'identity-obj-proxy'
  },
};

4. Create jest.setup.js:
require('@testing-library/jest-dom');
require("jest-fetch-mock").enableMocks();

5. Create global.d.ts:
declare const fetchMock: import('jest-fetch-mock').JestFetchMock;

6. Configure tsconfig.json:
"jsx": preserve -> "jsx": "react-jsx"
"include": ["next-env.d.ts", "**/*.ts", "**/*.tsx", ".next/types/**/*.ts"] ->
"include": ["next-env.d.ts", "**/*.ts", "**/*.tsx", ".next/types/**/*.ts, global.d.ts"],
